1. 计算给定整数序列的平均数（mean）、中位数（mid）和众数（mode）

以下是计算平均数、中位数和众数的代码实现：

use std::collections::HashMap;
use rand::Rng;
fn main{
    question1();
}

fn question1() {
    // 生成一个包含 20 个 1~9 之间随机数的 Vec
    let mut vec = Vec::new();
    let mut rng = rand::thread_rng();
    for _ in 0..20 {
        vec.push(rng.gen_range(1..10));
    }
    let vec_len = vec.len();
    println!("Generated vec: {:?}", &vec);

    // 计算总和，用于求平均数
    let mut sum: u32 = 0;

    // 用 HashMap 统计每个元素出现的次数，并寻找众数
    let mut map = HashMap::new();
    let mut mode: u32 = 0;
    let mut mode_count: u32 = 0;

    // 遍历数组，统计出现次数
    for &e in &vec {
        let count = map.entry(e).or_insert(0); // 获取该数字的出现次数，默认插入 0
        *count += 1; // 计数 +1
        sum += e; // 计算总和

        // 记录出现最多的元素（众数）
        if *count > mode_count {
            mode_count = *count;
            mode = e;
        }
    }

    // 计算平均值
    let avg: f32 = sum as f32 / vec_len as f32;

    // 对 Vec 进行排序
    sort(&mut vec);

    // 计算中位数（mid）
    let mid: f32 = if vec_len % 2 == 0 {
        // 若数组长度为偶数，中位数是中间两个数的平均值
        (vec[vec_len / 2 - 1] + vec[vec_len / 2]) as f32 / 2.0
    } else {
        // 若数组长度为奇数，中位数是中间的那个数
        vec[vec_len / 2] as f32
    };

    println!("avg: {}, mid: {}, mode: {}", avg, mid, mode);
}

// 实现一个简单的冒泡排序（可以使用 `vec.sort_unstable()` 代替）
fn sort(vec: &mut Vec<u32>) {
    for i in 0..vec.len() {
        for j in i + 1..vec.len() {
            if vec[i] > vec[j] {
                vec.swap(i, j); // 使用 Rust 提供的 `swap()` 方法替代手动交换
            }
        }
    }
    println!("Sorted Vec: {:?}", vec);
}

2. 将字符串转换为 Pig Latin

此练习题要求将字符串转换为 Pig Latin。具体规则为：每个单词的第一个辅音字母被移动到单词的结尾并增加 “ay”，元音字母开头的单词在结尾增加 “hay”。

fn main{
    question2();
}
fn question2() {
    let words = String::from("first");  // 要转换的单词
    let vowels = ['a', 'e', 'i', 'o', 'u'];  // 元音字母数组

    // 检查单词的第一个字符是否是元音字母
    if let Some(first_char) = words.chars().next() {
        if vowels.contains(&first_char) {
            // 如果第一个字符是元音字母，则直接在单词后加上 -hay
            let result = format!("{}-hay", words);
            println!("{}", result);
            return;
        }
    }

    // 如果第一个字符是辅音字母，则开始遍历字符
    let mut index = 0;
    for c in words.chars() {
        if vowels.contains(&c) {
            // 找到第一个元音字母
            let (front, end) = words.split_at(index);  // 将单词拆分为辅音部分和元音部分
            let result = format!("{}-{}ay", end, front);  // 拼接新的 Pig Latin 格式
            println!("{}", result);  // 输出转换后的结果
            return;
        }
        index += c.len_utf8();  // 更新索引，按照字符的字节长度移动
    }
}

3. 使用哈希 map 和 vector，创建一个文本接口来允许用户向公司的部门中增加员工的名字

此练习通过命令行接口，允许用户向公司的部门中添加员工的名字，并显示员工列表。

use std::collections::HashMap;
use std::io::{self, Write};

fn main{
    question3();
}

fn question3() {
    // 初始化一个 HashMap 来存储部门和员工
    let mut map: HashMap<String, Vec<String>> = HashMap::new();

    loop {
        // 打印提示信息，要求用户输入命令
        println!("Please input a command! (e.g., 'Add Sally to Engineering' or 'List Engineering'):");

        // 创建一个可变字符串，用来存储用户的输入
        let mut input = String::new();

        // 刷新标准输出流，确保提示信息能及时显示
        io::stdout().flush().unwrap();

        // 读取用户输入
        io::stdin().read_line(&mut input).unwrap();

        // 去掉输入字符串两端的空白字符
        let input = input.trim();

        // 如果用户输入 'q'，退出程序
        if input.to_lowercase() == "q" {
            println!("Goodbye============================================================================");
            break; // 退出循环
        }

        // 处理 'Add' 命令（添加员工到指定部门）
        if input.starts_with("Add") {
            // 将输入按空格分割成命令和参数
            let commands: Vec<&str> = input.split_whitespace().collect();

            // 如果命令格式正确（至少包含四个部分：Add、姓名、to、部门）
            if commands.len() >= 4 {
                let name = commands[1]; // 获取员工姓名
                let department = commands[3]; // 获取部门名称

                // 将员工添加到指定的部门
                map.entry(department.to_string()) // 获取部门的条目
                    .or_insert_with(Vec::new)     // 如果部门不存在，则初始化为空 Vec
                    .push(name.to_string());      // 将员工姓名添加到部门的员工列表中

                // 打印成功信息
                println!("Added {} to {}.============================================================================", name, department);
            } else {
                // 如果命令格式不正确，打印提示信息
                println!("Invalid format. Use 'Add <name> to <department>'============================================================================");
            }
        } 
        // 处理 'List' 命令（列出部门的员工）
        else if input.starts_with("List") {
            // 将输入按空格分割成命令和参数
            let commands: Vec<&str> = input.split_whitespace().collect();

            // 如果命令包含部门名，查询该部门的员工
            if commands.len() == 2 {
                // 查询指定部门的员工
                if let Some(employees) = map.get(commands[1]) {
                    // 如果部门存在，打印员工列表
                    println!("Employees in {}: {:?}", commands[1], employees);
                } else {
                    // 如果部门不存在，打印部门没有员工的提示
                    println!("No employees found in department '{}'.============================================================================",  commands[1]);
                }
            } 
            // 如果命令没有包含部门名，列出所有部门和员工
            else if commands.len() == 1 {
                // 如果没有部门，说明没有员工
                if map.len() == 0 {
                    println!("No departments found.============================================================================");
                } else {
                    // 遍历所有部门及其员工，打印每个部门的员工列表
                    for (department, employees) in &map {
                        println!("{}:{:?}", department, employees);
                    }
                }
            } else {
                // 如果命令格式不正确，打印提示信息
                println!("Invalid format. Use 'List <department>'============================================================================");
            }
        } 
        // 处理其他无效的命令格式
        else {
            println!("Invalid format. Use 'Add <name> to <department>'============================================================================");
        }
    }
}
