多线程事务的最终解决方案

Created2020-03-18|Updated2020-03-18|总结
|Post View:

1、导入jdbc-starter

2、导入aop-starter

 <dependency>
    <groupId>org.springframework.boot</groupId>
    <artifactId>spring-boot-starter-aop</artifactId>
 </dependency>

3、启动类标注注解

@MapperScan(basePackages = "xxx.xxx.xxx.mapper")
@SpringBootApplication
@EnableDubbo //注册服务
@EnableTransactionManagement //开启事务管理器
@EnableAspectJAutoProxy(exposeProxy = true) //开启切面自动代理 暴露代理对象
public class GmallPmsApplication {

    public static void main(String[] args) {
        SpringApplication.run(GmallPmsApplication.class, args);
    }

}

4、开启事务并用代理对象调方法

@Service
public class xxxImpl{
    @Transactional(propagation = Propagation.REQUIRED)
    public void function(xxx xxx) {
    //获取当前类的代理对象 用aop启动事务管理
      xxxImpl proxy = (xxxImpl) AopContext.currentProxy();
        //用代理对象调用方法
      proxy.a();
     }
    @Transactional(propagation = Propagation.REQUIRES_NEW)
      public void a(){

      }
}
